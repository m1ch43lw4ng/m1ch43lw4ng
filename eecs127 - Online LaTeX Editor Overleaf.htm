\section*{5. Matrix norms}
\begin{qparts}
\item We start with the definition in the problem
$$\norm{A}_1 = \max_{\vec{x} \not=
\vec{0}}\frac{\norm{A\vec{x}}_1}{\norm{\vec{x}}_1}$$
We rewrite it as suggested in the hint and then into a double summation form
$$\norm{A}_1 = \max_{\vec{x} \not= \vec{0}}\frac{\norm{\sum_{j =
1}^{m}\abs{x_j} \cdot \vec{a_j}}_1}{\norm{\vec{x}}_1}$$
$$ = \max_{\vec{x} \not= \vec{0}}\frac{\sum_{j = 1}^{m}\sum_{i =
1}^{n}\abs{x_j} \abs{a_{ij}}}{\norm{\vec{x}}_1}$$
$$ = \max_{\vec{x} \not= \vec{0}}\frac{\sum_{j = 1}^{m} \abs{x_j} \sum_{i =
1}^{n} \abs{a_{ij}}}{\norm{\vec{x}}_1}$$
WLOG, let us restrict $\norm{\vec{x}} = 1$
$$ = \max_{\norm{\vec{x}} = 1} \sum_{j = 1}^{m} \abs{x_j} \sum_{i = 1}^{n}
\abs{a_{ij}}$$
$$\norm{A}_1 \leq \sum_{j = 1}^{m} \abs{x_j} \max_{1 \leq j \leq m}(\sum_{i
= 1}^{n} \abs{a_{ij}})$$
$$ \leq \max_{1 \leq j \leq m}(\sum_{i = 1}^{n} \abs{a_{ij}})\norm{x}_1 $$
$$ \leq \max_{1 \leq j \leq m}(\sum_{i = 1}^{n} \abs{a_{ij}})$$
We can prove the equality by picking a vector $\vec{x}^*$ that is all 0s
except at the index $j^* = \text{argmax}_{1 \leq j \leq m}(\sum_{i = 1}^{n}
\abs{a_{ij}})$, where it has a value of one.
$$\sum_{j = 1}^{m} \abs{x_j^*} \sum_{i = 1}^{n} \abs{a_{ij^*}} =
x_j^*\max_{1 \leq j \leq m}(\sum_{i = 1}^{n} \abs{a_{ij}}) = \max_{1 \leq j
\leq n}(\sum_{i = 1}^{n} \abs{a_{ij}})$$
Because we know that $\max_{\norm{\vec{x}} = 1} \sum_{j = 1}^{m} \abs{x_j}
\sum_{i = 1}^{n} \abs{a_{ij}} \leq \max_{1 \leq j \leq m}(\sum_{i = 1}^{m}
\abs{a_{ij}})$ and we found an $\vec{x}^*$ that satisfies the equality, that
value is the max value of the LHS. Thus,
$$\boxed{\norm{A}_1 = \max_{1 \leq j \leq m}(\sum_{i = 1}^{m}
\abs{a_{ij}})}$$
\item
\end{qparts}
